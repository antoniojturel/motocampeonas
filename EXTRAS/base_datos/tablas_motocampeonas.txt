CREATE TABLE `cilindradas` (
  `ID_CILINDRADA` varchar(5) NOT NULL,
   PRIMARY KEY (`ID_CILINDRADA`)
);

CREATE TABLE `modalidades` (
  `ID_MODALIDAD` varchar(10) NOT NULL,
   PRIMARY KEY (`ID_MODALIDAD`)
);

CREATE TABLE `clasificaciones` (
  `ID_CLASIFICACION` varchar(15) NOT NULL,
   PRIMARY KEY (`ID_CLASIFICACION`)
);

CREATE TABLE `fechas` (
  `ID_FECHA` int(11) NOT NULL,
   PRIMARY KEY (`ID_FECHA`)
);

CREATE TABLE `paises` (
  `ID_PAIS` varchar(10) NOT NULL,
   PRIMARY KEY (`ID_PAIS`)
);

CREATE TABLE `categorias` (
  `ID_CATEGORIA` varchar(15) NOT NULL,
  `ID_MODALIDAD` varchar(10) NOT NULL,
  `ID_CILINDRADA` varchar(5) NOT NULL,
  `FECHA_COMIENZO` int(11) NOT NULL,
  `FECHA_FIN` int(11) DEFAULT NULL,
   PRIMARY KEY (`ID_CATEGORIA`),
   FOREIGN KEY (`ID_CILINDRADA`) REFERENCES `cilindradas` (`ID_CILINDRADA`),
   FOREIGN KEY (`ID_MODALIDAD`) REFERENCES `modalidades` (`ID_MODALIDAD`),
   FOREIGN KEY (`FECHA_COMIENZO`) REFERENCES `fechas` (`ID_FECHA`),
   FOREIGN KEY (`FECHA_FIN`) REFERENCES `fechas` (`ID_FECHA`)
);

CREATE TABLE `marcas` (
  `ID_MARCA` varchar(10) NOT NULL,
  `ID_PAIS` varchar(10) NOT NULL,
   PRIMARY KEY (`ID_MARCA`),
   FOREIGN KEY (`ID_PAIS`) REFERENCES paises (`ID_PAIS`) 
);

CREATE TABLE `usuarios` (
  `ID_USUARIO` int(11) NOT NULL AUTO_INCREMENT,
  `FECHA_REGISTRO` date NOT NULL,
  `NOMBRE_MOTE` varchar(15) DEFAULT NULL,
  `EMAIL` varchar(100) DEFAULT NULL,
  `PASSWORD` varchar(50) DEFAULT NULL,
  `ID_MARCA` varchar(10) DEFAULT NULL,
  `ID_PAIS` varchar(10) DEFAULT NULL,
   PRIMARY KEY (`ID_USUARIO`),
   FOREIGN KEY (`ID_MARCA`) REFERENCES `marcas` (`ID_MARCA`),
   FOREIGN KEY (`ID_PAIS`) REFERENCES paises (`ID_PAIS`)
);

CREATE TABLE `victorias` (
  `ID_VICTORIA` int(11) NOT NULL AUTO_INCREMENT,
  `ID_CATEGORIA` varchar(15) NOT NULL,
  `ID_CLASIFICACION` varchar(15) NOT NULL,
  `ID_FECHA` int(11) NOT NULL,
  `ID_MARCA` varchar(10) NOT NULL,
   PRIMARY KEY(ID_VICTORIA),
   FOREIGN KEY (`ID_MARCA`) REFERENCES `marcas` (`ID_MARCA`),
   FOREIGN KEY (`ID_CATEGORIA`) REFERENCES `categorias` (`ID_CATEGORIA`),
   FOREIGN KEY (`ID_FECHA`) REFERENCES `fechas` (`ID_FECHA`),
   FOREIGN KEY (`ID_CLASIFICACION`) REFERENCES `clasificaciones` (`ID_CLASIFICACION`)
);

CREATE TABLE `votos` (
  `ID_VOTO` int(11) NOT NULL AUTO_INCREMENT,
  `ID_CATEGORIA` varchar(15) NOT NULL,
  `ID_CLASIFICACION` varchar(15) NOT NULL,
  `ID_MARCA` varchar(10) NOT NULL,
  `ID_FECHA` int(11) NOT NULL,
   PRIMARY KEY(ID_VOTO),
   FOREIGN KEY (`ID_CATEGORIA`) REFERENCES `categorias` (`ID_CATEGORIA`),
   FOREIGN KEY (`ID_CLASIFICACION`) REFERENCES `clasificaciones` (`ID_CLASIFICACION`),
   FOREIGN KEY (`ID_MARCA`) REFERENCES `marcas` (`ID_MARCA`),
   FOREIGN KEY (`ID_FECHA`) REFERENCES `fechas` (`ID_FECHA`)
);
